CREATE TABLE question (
id SMALLINT UNSIGNED AUTO_INCREMENT,
student_id MEDIUMINT UNSIGNED,
top_eventNo MEDIUMINT UNSIGNED,
title VARCHAR(63),
content VARCHAR(500),
answer VARCHAR(500),
tagID SMALLINT UNSIGNED,
PRIMARY KEY (id),
FOREIGN KEY (tagID)
REFERENCES tag(id)
ON DELETE SET NULL,
INDEX (title)
)ENGINE=INNODB DEFAULT CHARSET=utf8;
