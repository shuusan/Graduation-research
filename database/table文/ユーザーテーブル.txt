CREATE TABLE user (
id MEDIUMINT UNSIGNED,
department_id TINYINT UNSIGNED,
name VARCHAR(63),
password VARCHAR(63),
grade TINYINT(10) UNSIGNED,
authority TINYINT(10) UNSIGNED,
mail VARCHAR(127),
PRIMARY KEY(id),
FOREIGN KEY (department_id)
REFERENCES department_cose(id)
ON UPDATE CASCADE ON DELETE SET NULL,
INDEX (id)
)ENGINE=INNODB;
